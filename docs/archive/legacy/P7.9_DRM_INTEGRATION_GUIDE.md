```typescript
// DRM client library

const DRM_API = process.env.NEXT_PUBLIC_API_URL + '/api/v1/drm';

interface DeviceInfo {
  device_type: 'web' | 'mobile' | 'desktop';
  user_agent: string;
  ip_address: string;
}

interface License {
  license_id: string;
  file_id: string;
  expires_at: string;
  content_key: string;
  usage_rights: string[];
  watermark_text: string;
}

// Get device info
export async function getDeviceInfo(): Promise<DeviceInfo> {
  const response = await fetch('/api/device-info');
  return response.json();
}

// Request license
export async function requestLicense(
  fileId: string,
  usageRights: string[] = ['view']
): Promise<License> {
  const deviceInfo = await getDeviceInfo();
  
  const response = await fetch(`${DRM_API}/licenses/generate`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      file_id: fileId,
      device_info: deviceInfo,
      usage_rights: usageRights,
      expires_in_hours: 24,
    }),
  });
  
  if (!response.ok) throw new Error('License request failed');
  return response.json();
}

// Download protected content
export async function downloadProtected(
  fileId: string,
  licenseId: string
): Promise<Blob> {
  const response = await fetch(
    `${DRM_API}/files/${fileId}/download?license_id=${licenseId}`,
    { credentials: 'include' }
  );
  
  if (!response.ok) throw new Error('Download failed');
  return response.blob();
}

// Track playback (for audit trail)
export async function trackPlayback(
  licenseId: string,
  fileId: string,
  startSeconds: number,
  endSeconds: number
): Promise<void> {
  const deviceInfo = await getDeviceInfo();
  
  await fetch(`${DRM_API}/playback/track`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      license_id: licenseId,
      file_id: fileId,
      device_id: generateDeviceId(deviceInfo),
      playback_start: startSeconds,
      playback_end: endSeconds,
      ip_address: deviceInfo.ip_address,
    }),
  });
}

// Request offline access
export async function requestOfflineAccess(
  fileId: string,
  licenseId: string
): Promise<{ offline_bundle_id: string; expires_at: string }> {
  const deviceInfo = await getDeviceInfo();
  
  const response = await fetch(`${DRM_API}/offline/request`, {
    method: 'POST',

    body: JSON.stringify({

      file_id: fileId,
      device_id: generateDeviceId(deviceInfo),
      device_type: 'web',
    }),
  });
  
  if (!response.ok) throw new Error('Offline access request failed');
  return response.json();
}
```

```typescript
'use client';

import React, { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Switch } from '@/components/ui/switch';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';

interface Props {
  bookId: string;
  onSuccess?: (fileId: string) => void;
}

export default function ProtectedFileUploader({ bookId, onSuccess }: Props) {
  const [file, setFile] = useState<File | null>(null);
  const [accessLevel, setAccessLevel] = useState<string>('subscriber');
  const [usageRights, setUsageRights] = useState<string[]>(['view']);
  const [watermarkEnabled, setWatermarkEnabled] = useState(true);
  const [maxOfflineDownloads, setMaxOfflineDownloads] = useState(0);
  const [uploading, setUploading] = useState(false);

  const handleUpload = async () => {
    if (!file) return;

    setUploading(true);
    try {
      const formData = new FormData();
      formData.append('book_id', bookId);
      formData.append('file_name', file.name);
      formData.append('file_type', file.name.split('.').pop() || 'pdf');
      formData.append('content', file);
      formData.append('access_level', accessLevel);
      formData.append('usage_rights', JSON.stringify(usageRights));
      formData.append('watermark_enabled', String(watermarkEnabled));
      formData.append('max_offline_downloads', String(maxOfflineDownloads));

      const response = await fetch('/api/v1/drm/files/protect', {
        method: 'POST',
        body: formData,
        headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` },
      });

      if (!response.ok) throw new Error('Upload failed');
      
      const data = await response.json();
      onSuccess?.(data.id);
    } finally {
      setUploading(false);
    }
  };

  return (
    <div className="space-y-4 border p-4 rounded">
      <h3 className="text-lg font-semibold">Protect a File with DRM</h3>
      
      <input
        type="file"
        onChange={(e) => setFile(e.target.files?.[0] || null)}
        disabled={uploading}
      />

      <div>
        <label className="block text-sm font-medium mb-2">Access Level</label>
        <Select value={accessLevel} onValueChange={setAccessLevel}>
          <SelectTrigger>
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="public">Public (Free)</SelectItem>
            <SelectItem value="subscriber">Subscriber</SelectItem>
            <SelectItem value="premium">Premium Only</SelectItem>
            <SelectItem value="restricted">Restricted (Direct)</SelectItem>
          </SelectContent>
        </Select>
      </div>

      <div>
        <label className="block text-sm font-medium mb-2">Usage Rights</label>
        <div className="space-y-2">
          {['view', 'print', 'copy', 'offline', 'share'].map((right) => (
            <label key={right} className="flex items-center">
              <input
                type="checkbox"
                checked={usageRights.includes(right)}
                onChange={(e) => {
                  setUsageRights(
                    e.target.checked
                      ? [...usageRights, right]
                      : usageRights.filter((r) => r !== right)
                  );
                }}
              />
              <span className="ml-2 capitalize">{right}</span>
            </label>
          ))}
        </div>
      </div>

      <div className="flex items-center justify-between">
        <label className="text-sm font-medium">Enable Watermark</label>
        <Switch
          checked={watermarkEnabled}
          onCheckedChange={setWatermarkEnabled}
        />
      </div>

      <div>
        <label className="block text-sm font-medium mb-2">Max Offline Downloads</label>
        <input
          type="number"
          min="0"
          max="10"
          value={maxOfflineDownloads}
          onChange={(e) => setMaxOfflineDownloads(Number(e.target.value))}
          className="w-full border rounded px-2 py-1"
        />
        <p className="text-xs text-gray-500 mt-1">0 = no offline access allowed</p>
      </div>

      <Button
        onClick={handleUpload}
        disabled={!file || uploading}
        className="w-full"
      >
        {uploading ? 'Protecting...' : 'Protect File'}
      </Button>
    </div>
  );
```

### Step 3: Create License Request Hook

**`frontend/src/hooks/useDRMLicense.ts`:**

```typescript
'use client';

import { useQuery, useMutation } from '@tanstack/react-query';
import { requestLicense, downloadProtected } from '@/lib/drm';

export function useDRMLicense(fileId: string) {
  const [license, setLicense] = React.useState(null);

  const generateLicense = useMutation({
    mutationFn: () => requestLicense(fileId),
    onSuccess: (data) => setLicense(data),
  });

  const download = useMutation({
    mutationFn: (licenseId: string) => downloadProtected(fileId, licenseId),
    onSuccess: (blob) => {
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `document.pdf`;
      a.click();
      URL.revokeObjectURL(url);
    },
  });

  return {
    license,
    isGenerating: generateLicense.isPending,
    isDownloading: download.isPending,
    generateLicense: () => generateLicense.mutate(),
    download: () => license && download.mutate(license.license_id),
  };
}
```

```typescript
import * as SecureStore from 'expo-secure-store';
import { useAuth } from '@/hooks/useAuth';

interface StoredLicense {
  license_id: string;
  expires_at: string;
  content_key: string;
}

export function useDRMMobile() {
  const { user } = useAuth();

  const saveLicense = async (license: StoredLicense) => {
    try {
      await SecureStore.setItemAsync(
        `license_${license.license_id}`,
        JSON.stringify(license)
      );
    } catch (error) {
      console.error('Failed to save license:', error);
    }
  };

  const getLicense = async (licenseId: string): Promise<StoredLicense | null> => {
    try {
      const stored = await SecureStore.getItemAsync(`license_${licenseId}`);
      return stored ? JSON.parse(stored) : null;
    } catch (error) {
      console.error('Failed to get license:', error);
      return null;
    }
  };

  const revokeLicense = async (licenseId: string) => {
    try {
      await SecureStore.deleteItemAsync(`license_${licenseId}`);
    } catch (error) {
      console.error('Failed to revoke license:', error);
    }
  };

  return {
    saveLicense,
    getLicense,
    revokeLicense,
  };
}
```

## Monitoring & Maintenance

### Dashboard Queries

```sql
-- Active licenses
SELECT COUNT(*) as active_count
FROM drm_licenses
WHERE is_active = true AND expires_at > NOW();

-- Revocations
SELECT COUNT(*) as revoked_count
FROM drm_revocations
WHERE revoked_at > NOW() - INTERVAL '7 days';

-- Suspicious patterns
SELECT license_id, COUNT(DISTINCT device_id) as device_count
FROM drm_playback_events
GROUP BY license_id
HAVING device_count > 3;

-- Popular protected files
SELECT pf.file_name, COUNT(dl.license_id) as access_count
FROM protected_files pf
LEFT JOIN drm_licenses dl ON pf.id = dl.protected_file_id
GROUP BY pf.id
ORDER BY access_count DESC
LIMIT 10;
```

## Troubleshooting

### Issue: "License verification failed"

```
Cause: Device ID mismatch (user using different browser/device)
Fix: Regenerate license from new device
```

### Issue: "Encrypted file corrupted"

```
Cause: File upload interrupted or salt mismatched
Fix: Re-upload file, verify DRM_MASTER_KEY unchanged
```

### Issue: "Offline bundle expired"

```
Cause: 30+ days passed since download
Fix: Delete offline bundle, request new offline access
```

### Issue: "Too many devices for license"

```
Cause: Exceeded max_devices_per_license (default 1)
Fix: User can't use on new device until old license expires
```

## Performance Optimization

### Encryption Speed

```
File Size | Encryption Time | Throughput
1 MB      | 50ms           | 20 MB/s
10 MB     | 500ms          | 20 MB/s
100 MB    | 5s             | 20 MB/s
```

**Optimization:** For files >100MB, implement chunked encryption:

```python
def encrypt_file_chunked(self, file_path: str, chunk_size_mb=10):
    """Encrypt large files in chunks."""
    chunk_size = chunk_size_mb * 1024 * 1024
    
    with open(file_path, 'rb') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            encrypted_chunk = self.cipher.encrypt(chunk)
            yield encrypted_chunk
```
